import random
from datetime import date, timedelta
from app import db, app
from models import User, RoomAllocation, Storefront, PilgrimBooking

def seed_data():
    with app.app_context():
        # 1. Clear existing test data (Optional)
        # db.drop_all() 
        # db.create_all()

        print("Seeding PHX Exchange with 2026 Mock Data...")

        # 2. Create Sample Agents (if none exist)
        agent_names = ["Al-Safar Travel", "Global Hajj Group", "Nusuk Partners"]
        agents = []
        for name in agent_names:
            slug = name.lower().replace(" ", "-")
            user = User(username=slug, role='AGENT', wallet_balance=random.uniform(5000, 20000))
            db.session.add(user)
            db.session.flush() # Get ID
            
            store = Storefront(agent_id=user.id, slug=slug, agency_name=name, markup_percent=12.5)
            db.session.add(store)
            agents.append(user)

        # 3. Create Room Allocations
        rooms = []
        for i in range(10):
            room = RoomAllocation(
                hotel_name=random.choice(["Fairmont Makkah", "Hilton Suite", "Pullman Zamzam"]),
                room_type=random.choice(["Triple", "Quad", "Double"]),
                base_price=random.randint(1500, 4500),
                status='AVAILABLE',
                agent_id=random.choice(agents).id
            )
            db.session.add(room)
            rooms.append(room)
        
        db.session.flush()

        # 4. Create Mock Pilgrim Bookings
        countries = ["IDN", "PAK", "TUR", "GBR", "USA", "MYS"]
        for i in range(15):
            target_agent = random.choice(agents)
            target_room = random.choice([r for r in rooms if r.agent_id == target_agent.id])
            
            # Markup Logic
            final_price = target_room.base_price * 1.125
            
            # Status Logic (Mix of Pending and Issued)
            v_status = random.choice(['PENDING', 'ISSUED'])
            v_num = f"V26-{random.randint(10000, 99999)}" if v_status == 'ISSUED' else None

            booking = PilgrimBooking(
                allocation_id=target_room.id,
                agent_id=target_agent.id,
                full_name=f"Pilgrim {i+1} Example",
                citizenship=random.choice(countries),
                passport_number=f"A{random.randint(1000000, 9999999)}",
                dob=date(1980, 1, 1) + timedelta(days=random.randint(0, 10000)),
                passport_expiry=date(2027, 6, 1), # Compliant with 2026 rules
                nusuk_id=str(random.randint(1000000000, 9999999999)),
                visa_status=v_status,
                visa_number=v_num,
                final_price_paid=final_price
            )
            db.session.add(booking)

        db.commit()
        print("Success! 10 Rooms and 15 Pilgrim Bookings created.")

if __name__ == "__main__":
    seed_data()
